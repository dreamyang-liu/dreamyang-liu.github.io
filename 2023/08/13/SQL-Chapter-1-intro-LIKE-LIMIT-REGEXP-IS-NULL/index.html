from customers

-- 第1页：1-3
-- 第2页：4-6
-- 第3页：7-9
limit 6, 3
-- 告诉sql跳过前6条记录并执行3条记录
	-- 在这种情况下，7-9

-- limit子句也应该放在最后
SELECT *
FROM customers
order by points DESC
limit 3;
